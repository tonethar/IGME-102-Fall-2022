### II-C. Loops
- See *Functions and Loops* PDF in myCourses

**Loop through numbers**
```js
for(let i=0;i<5;i++){
  console.log(i);
}
```

**Loop through array**
```js
const animals = ["dog","cat","fish"];
for(let i=0;i<animals.length;i++){
  console.log(animals[i]);
}
```

**Draw some circles**
```js
for(let i=0; i<5;i++){
  circle((i* 60) + 50,200,50);
}
```

**Draw the same circles, but only use the `for` loop variable**
```js
for(let x=50; x<300;x+=60){
  circle(x,200,50);
}
```

**Draw the same circles, but with a `while` loop**
```js
let x = 50;
while (x < 300){
  circle(x,200,50);
  x += 60;
}
```

**Randomly sized ellipses**
```js
let y = 30;
while (y < 300) {
  ellipse (50, y , random(20, 50), random(20, 50));
  y+=50;
}
```

**Get a random string from an array**
```js
const colors = ["red","green","blue"];
let randColor = colors[int(random(colors.length))];
console.log(randColor);
```

**Nested for loops**
```js
for (let n = 1; n < 4; n++) {
      console.log("OUTER Loop " + n);

      //print 3 statements in between 
      for (let i = 1; i < 4; i++)
      {
           console.log("Inner Loop " + i);
      }
  }
```

### III-C. JS Classes
- Review Goose Demo
- JS Classes

**Minimal class**
```js
class Nada{}  // does nothing
let nada = new Nada();
console.log(nada);
```

**Class with constructor & default properties**
```js
class Cat{
  constructor(){
    this.species = "Cat";
  }
} // end class Cat

let cat = new Cat();
console.log(cat);
```

**Class with constructor parameters**
```js
class Dog{
  constructor(name){
    this.name = name;
    this.species = "Dog";
  }
} // end class Dog

let dog = new Dog("Wags");
console.log(dog);
```

**Class with methods**
```js
class Sprite{
  constructor(x,y,width,height,color){
    this.x = x;
    this.y = y;
    this.width = width;
    this.height = height;
    this.color = color;
  }
  
  display(){
    fill(this.color);
    rect(this.x,this.y,this.width,this.height);
  }
} // end class Sprite

let sprite1 = new Sprite(10,10,50,50,"red");
console.log(sprite1);
sprite1.display();

let sprite2 = new Sprite(10,100,50,50,"green");
console.log(sprite2);
sprite2.display();
```
